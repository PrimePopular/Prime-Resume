        function updatePremiumButtons() {
            const isPremium = localStorage.getItem('primePremiumToken') || 
                            localStorage.getItem('devMode') === 'true';
            
            document.querySelectorAll('[data-premium-only]').forEach(btn => {
                if (isPremium) {
                    btn.disabled = false;
                    btn.style.opacity = '1';
                } else {
                    btn.disabled = true;
                    btn.style.opacity = '0.6';
                }
            });
        }

        // Generate premium token
        function generatePremiumUrl() {
            const token = 'premium_' + Date.now() + '_' + Math.random().toString(36).substring(2, 18);
            const baseUrl = window.location.origin + window.location.pathname.replace('success.html', '');
            const premiumUrl = `${baseUrl}build.html#premium=${token}`;
            
            document.getElementById('premiumUrl').textContent = premiumUrl;
            localStorage.setItem('primePremiumToken', token);
            
            // Update builder link
            const builderLink = document.querySelector('a[href="build.html"]');
            builderLink.href = premiumUrl;
        }
        
        // Copy URL to clipboard
        function copyUrl() {
            const url = document.getElementById('premiumUrl').textContent;
            navigator.clipboard.writeText(url).then(() => {
                alert('‚úÖ Premium link copied! Save it somewhere safe.');
            });
        }
        
        // Initialize
        window.onload = generatePremiumUrl;
